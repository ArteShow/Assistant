Add Task
    curl -X POST http://localhost:8082/task/add -H "Content-Type: application/json" -d "{\"title\": \"My Task\", \"status\": \"open\", \"description\": \"This is a test task\", \"user_id\": 1}"

Delete Task
    curl -X POST http://localhost:8082/task/delete -H "Content-Type: application/json" -d "{\"task_ID\": 1, \"user_ID\": 1}"
