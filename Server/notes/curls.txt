# ---------- TASKS ----------
# Add a new task (user_id auto from token, money = worth of task)
curl -X POST http://localhost:8082/task/add -H "Authorization: Bearer " -H "Content-Type: application/json" -d "{\"title\": \"My Task\", \"status\": \"In Progress\", \"description\": \"This is a test task\", \"money\": 100}"

# Delete a task by ID
curl -X POST http://localhost:8082/task/delete -H "Authorization: Bearer " -H "Content-Type: application/json" -d "{\"task_ID\": 1}"

# Get ALL tasks in the database (admin/global view)
curl -X GET http://localhost:8082/task/getTasksList -H "Authorization: Bearer " -H "Content-Type: application/json"

# Get all tasks for the logged-in user
curl -X POST http://localhost:8082/task/getUsersTaskList -H "Authorization: Bearer " -H "Content-Type: application/json"

# Get a single task by its ID (only if it belongs to logged-in user)
curl -X POST http://localhost:8082/task/getTaskByID -H "Authorization: Bearer " -H "Content-Type: application/json" -d "{\"task_id\": 1}"

# Change the status of a task
curl -X POST http://localhost:8082/task/editTasksStatus -H "Authorization: Bearer " -H "Content-Type: application/json" -d "{\"task_id\": 1, \"status\": \"Finished\"}"

# ---------- MONEY ----------
# Set a new savings goal
curl -X POST http://localhost:8082/money/setGoal -H "Authorization: Bearer " -H "Content-Type: application/json" -d "{\"goal\": 13500}"

# Add money towards the goal
curl -X POST http://localhost:8082/money/addMoney -H "Authorization: Bearer " -H "Content-Type: application/json" -d "{\"amount\": 100}"

# Get current savings stats
curl -X GET http://localhost:8082/money/getStats -H "Authorization: Bearer " -H "Content-Type: application/json"

# ---------- AUTH ----------
# Register a new user
curl -X POST http://localhost:8082/registration -H "Content-Type: application/json" -d "{\"username\": \"ArteShow\", \"password\": \"123\"}"

# Login (returns JWT token)
curl -X POST http://localhost:8082/login -H "Content-Type: application/json" -d "{\"username\": \"ArteShow\", \"password\": \"123\"}"
